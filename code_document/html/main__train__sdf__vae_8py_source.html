69# Define the loss function
70def lossfunc(sdf,sdf_hat,iso,iso_hat,mu,logvar,beta):
71 """
72 Computes the Variational Autoencoder (VAE) loss function, combining reconstruction loss and KL divergence.
73 Args
74 Returns:
75 torch.Tensor: The computed loss value.
76 """
77 sdf_loss = F.mse_loss(sdf_hat, sdf,reduction = 'sum')
78 iso_loss = F.mse_loss(iso_hat, iso,reduction = 'sum')*40**3
79 # iso_loss = F.mse_loss(iso_hat, iso,reduction = 'sum')*30**3 # for another case
80 recons_loss = sdf_loss+ iso_loss
81 kl_loss = -0.5 * torch.sum(1 + logvar - mu.pow(2) - logvar.exp())
82 # kl_loss = torch.mean(-0.5 * torch.sum(1 + logvar - mu.pow(2) - logvar.exp(), dim = 1), dim = 0)
83 return recons_loss + beta* kl_loss
